Implementation approach

Uses existing Bootstrap shown.bs.tab event + per-tab loaded flag

// In reports.html — add per-tab lazy-load hook
document.querySelectorAll('[data-bs-toggle="tab"]').forEach(tab => {
  tab.addEventListener('shown.bs.tab', e => {
    if (!tabLoaded[tabId]) { fetchTabData(tabId); tabLoaded[tabId] = true; }
  });
});
